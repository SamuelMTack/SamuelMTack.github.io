
import React, { useState, useEffect } from 'react';
import { GameScore, User } from '@/entities/all';
import GameCanvas from '../components/game/GameCanvas';
import GameStats from '../components/game/GameStats';
import { Card, CardContent, CardHeader, CardTitle } from "@/components/ui/card";
import { Gamepad2, Sparkles } from "lucide-react";

export default function GamePage() {
    const [gameStats, setGameStats] = useState([]);
    const [user, setUser] = useState(null);
    const [startTime] = useState(Date.now());

    useEffect(() => {
        loadData();
    }, []);

    const loadData = async () => {
        try {
            const currentUser = await User.me();
            setUser(currentUser);
            const stats = await GameScore.list('-score', 10);
            setGameStats(stats.filter(s => s.created_by === currentUser.email));
        } catch (error) {
            console.log('User not logged in or error loading data');
        }
    };

    const handleScoreUpdate = async (score) => {
        // Update in real-time, will save when game ends
    };

    const handleGameOver = async (finalScore, level) => {
        try {
            if (user) {
                const timePlayedSeconds = Math.floor((Date.now() - startTime) / 1000);
                const gemsCollected = Math.floor(finalScore / 100); // Each gem is 100 points
                
                await GameScore.create({
                    score: finalScore,
                    level: level,
                    gems_collected: gemsCollected,
                    time_played: timePlayedSeconds
                });
                
                // Reload stats
                await loadData();
            }
        } catch (error) {
            console.error('Error saving game score:', error);
        }
    };

    return (
        <div className="min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-slate-900 p-4 md:p-8">
            <div className="max-w-6xl mx-auto space-y-8">
                {/* Header */}
                <div className="text-center space-y-4">
                    <div className="flex items-center justify-center gap-3">
                        <div className="w-12 h-12 bg-gradient-to-br from-indigo-500 to-purple-600 rounded-xl flex items-center justify-center">
                            <Gamepad2 className="w-7 h-7 text-white" />
                        </div>
                        <div>
                            <h1 className="text-4xl font-bold bg-gradient-to-r from-indigo-400 to-purple-400 bg-clip-text text-transparent">
                                Crystal Leap
                            </h1>
                            <p className="text-slate-400">An elegant platformer adventure</p>
                        </div>
                    </div>
                    
                    <div className="flex items-center justify-center gap-2 text-slate-300">
                        <Sparkles className="w-4 h-4 text-yellow-400" />
                        <span className="text-sm">Collect gems, master the platforms, reach new heights</span>
                        <Sparkles className="w-4 h-4 text-yellow-400" />
                    </div>
                </div>

                {/* Game Canvas */}
                <div className="flex justify-center">
                    <div className="w-full max-w-4xl">
                        <GameCanvas 
                            onScoreUpdate={handleScoreUpdate}
                            onGameOver={handleGameOver}
                        />
                    </div>
                </div>

                {/* Game Instructions */}
                <Card className="bg-slate-800/50 border-slate-700 backdrop-blur">
                    <CardHeader>
                        <CardTitle className="text-white text-lg">How to Play</CardTitle>
                    </CardHeader>
                    <CardContent className="space-y-3">
                        <div className="grid md:grid-cols-3 gap-6 text-slate-300">
                            <div className="space-y-2">
                                <h4 className="font-semibold text-indigo-400">Movement & Combat</h4>
                                <ul className="space-y-1 text-sm">
                                    <li>• <kbd className="px-2 py-1 bg-slate-700 rounded text-xs">A/D</kbd> or <kbd className="px-2 py-1 bg-slate-700 rounded text-xs">←/→</kbd> to move left/right</li>
                                    <li>• <kbd className="px-2 py-1 bg-slate-700 rounded text-xs">W/Space/↑</kbd> to jump</li>
                                    <li>• Jump on <span className="text-red-400">enemies</span> to defeat them. Don't let them touch you!</li>
                                </ul>
                            </div>
                            <div className="space-y-2">
                                <h4 className="font-semibold text-pink-400">Special Platforms</h4>
                                <ul className="space-y-1 text-sm">
                                    <li>• <span className="text-pink-400">Pink Platforms:</span> Super bounce!</li>
                                    <li>• <span className="text-blue-400">Blue Platforms:</span> Speed boost!</li>
                                    <li>• <span className="text-emerald-400">Green Platforms:</span> Normal.</li>
                                </ul>
                            </div>
                            <div className="space-y-2">
                                <h4 className="font-semibold text-emerald-400">Objectives</h4>
                                <ul className="space-y-1 text-sm">
                                    <li>• Collect golden gems for points (100 each)</li>
                                    <li>• Defeat enemies for bonus points (250 each)</li>
                                    <li>• Reach the end to advance to harder levels</li>
                                </ul>
                            </div>
                        </div>
                    </CardContent>
                </Card>

                {/* Stats */}
                {user && (
                    <div className="space-y-4">
                        <h2 className="text-2xl font-bold text-white text-center">Your Performance</h2>
                        <GameStats stats={gameStats} />
                    </div>
                )}

                {!user && (
                    <Card className="bg-slate-800/50 border-slate-700 backdrop-blur">
                        <CardContent className="text-center p-6">
                            <p className="text-slate-400 mb-4">Sign in to track your high scores and compete with others!</p>
                            <button
                                onClick={() => User.login()}
                                className="px-6 py-2 bg-indigo-600 hover:bg-indigo-700 text-white rounded-lg transition-colors"
                            >
                                Sign In to Save Progress
                            </button>
                        </CardContent>
                    </Card>
                )}
            </div>
        </div>
    );
}
